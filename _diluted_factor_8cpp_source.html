80  std::vector<DilutedFactor> const &right_vec) {
82  auto constexpr rnd_vec_count = 5;
83 
84  std::vector<DilutedFactor> result_vec;
85

87  auto const inner_rnd_id = left.ric.second;
88 
91  std::vector<Eigen::MatrixXcd> right_sums(
92  rnd_vec_count, Eigen::MatrixXcd::Zero(dilE * dilD, dilE * dilD));
93 
94  for (auto const &right : right_vec) {
95  // We may only multiply two diluted factors if the new outer indices are different
96  // and the inner indices are the same.
97  bool const is_allowed =
98  left.ric.second == right.ric.first && left.ric.first != right.ric.second;
99  if (!is_allowed) {
100  continue;
101  }
102 
103  right_sums[right.ric.second] += right.data;
104  }
105 
106  for (int right_outer = 0; right_outer != right_sums.size(); ++right_outer) {
107  // There will not be an element for this case since that is forbidden.
108  if (right_outer == inner_rnd_id) {
109  continue;
110  }
111  result_vec.emplace_back(left.data * right_sums[right_outer],
112  right_vec[0].left_Gamma,
113  std::make_pair(left.ric.first, right_outer),

115  }
116  }
